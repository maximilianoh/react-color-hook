SwatchesColor without color

```js
import SwatchesColor from './SwatchesColor.jsx';

```


SwatchesColor color
```js
import SwatchesColor from './SwatchesColor.jsx';
```

```js
import SwatchesColor from './SwatchesColor.jsx';
<div>

    <SwatchesColor active color="#EEC954"/>
    <SwatchesColor last color="#5481EE"/>
</div>
```
